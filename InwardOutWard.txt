CREATE TABLE RackMst(
   RackId serial PRIMARY KEY  NOT NULL,
   BranchId integer NOT NULL references BranchMst(BranchId),
   RackNo varchar(200) NOT NULL,
   CreationDt timestamp,
   Status Char(1)  CHECK(STATUS IN ('A', 'D')) default 'A' 
);

CREATE TABLE ShelfMst(
   ShelfId serial PRIMARY KEY  NOT NULL,
   BranchId integer NOT NULL references BranchMst(BranchId),
   RackId integer NOT NULL references RackMst(RackId),
   ShelfNo varchar(200) NOT NULL,
   CreationDt timestamp,
   Status Char(1)  CHECK(STATUS IN ('A', 'D')) default 'A'  
);

CREATE TABLE CompartMst(
   CompartId serial PRIMARY KEY  NOT NULL,
   BranchId integer NOT NULL references BranchMst(BranchId),
   RackId integer   NOT NULL references RackMst(RackId),
   ShelfId integer NOT NULL references ShelfMst(ShelfId),
   CompNo varchar(200) NOT NULL,
   CreationDt timestamp,
   Status Char(1)  CHECK(STATUS IN ('A', 'D')) default 'A'
);

CREATE TABLE BranchMst(
   BranchId serial PRIMARY KEY  NOT NULL,
   BranchNm varchar(200) UNIQUE NOT NULL,
   BranchAddress varchar(200) NOT NULL,
   CreationDt timestamp,
   Status Char(1)  CHECK(STATUS IN ('A', 'D')) default 'A' 
);


CREATE TABLE FileInOutTrans(
   TransId serial PRIMARY KEY  NOT NULL,
   UserId bigint NOT NULL references UserMst(UserId),
   FileId bigint NOT NULL references FileMst(FileId),
   BranchId integer NOT NULL references BranchMst(BranchId),
   RackId integer NOT NULL references RackMst(RackId ),
   ShelfId integer NOT NULL references ShelfMst(ShelfId),
   CompartId integer NOT NULL references CompartMst(CompartId),
   TXNType char(1) NOT NULL CHECK(STATUS IN ('I', 'O')),
   TXNDate timestamp NOT NULL,
   AuthorityNM varchar(200) NOT NULL,
   Remark varchar(200) NOT NULL,
   RDate timestamp NOT NULL,
   RTime TIME NOT NULL,
   RMobile numeric(12,0) NOT NULL
 );
